#include <stdio.h>
#include <conio.h> // inclure cette librairie pour la fonction system dans le main
#include <windows.h> // inclure cette librairie pour les 2 sous-programmes qui suivent


/*
 * Nom du sous-programme de la librairie windows.h à copier : gotoligcol
 * Rôle : positionne le curseur à l'écran
 * Paramètres : lig pour la ligne et col pour la colonne du curseur à 'écran
 */
void gotoligcol( int lig, int col )
{
    // ressources
    COORD mycoord;
    mycoord.X = col;
    mycoord.Y = lig;
    SetConsoleCursorPosition( GetStdHandle( STD_OUTPUT_HANDLE ), mycoord );
}


/*
 * Nom du sous-programme de la librairie windows.h à copier  : Color
 * Rôle : change la couleur du texte dans la console, ainsi que la couleur du fond pour la ligne
 * Paramètres : couleurDuTexte pour la couleur du texte et couleurDeFond pour la couleur du fond
 */
void Color(int couleurDuTexte,int couleurDeFond)
{
    HANDLE H=GetStdHandle(STD_OUTPUT_HANDLE);
    SetConsoleTextAttribute(H,couleurDeFond*16+couleurDuTexte);
}



int main() {
    int option;

    // positionner le curseur à la ligne 5 colonne 20 de curseur avant d'afficher le message
    gotoligcol(5,10);


    //On affiche le menu avec les différents choix qui peuvent etre choisit. On affiche ces choix là avec des couleurs 
    Color(7, 15);
    printf("Menu Principal\n\n");

    
    Color(7, 15);
    printf("1. commencer la partie (écrire 1)\n\n");

    
    Color(7, 15);
    printf("2. afficher aide (écrire 2)\n\n");

    
    Color(7, 15);
    printf("3. afficher score (écrire 3)\n\n");

    
    Color(7, 15);
    printf("4. quitter jeu (écrire 4)\n\n");

    
    Color(7, 15);
    printf("5. reprendre partie (écrire 5)\n\n");


    printf("Selectionner une option:\n ",);
    scanf("%d",&option);
    while (option =! 1 || option != 2 || option != 3 || option != 4 || option != 5 ){
        printf("Le choix n_est pas valide. Veuillez selectionner une option valide.\n");
        scanf("%d",&option);
    }
    switch(option) // on creer un switch qui va faire une certaine chose par rapport à ce que l'utilisateur choisi
    {
        case 1:
            CommencerPartie(); // s'il choisit le 1 alors on lance le programme du commencement du jeu
        case 2:

            return 1

        case 5:

            break;
    }
    return 0;
}
